Sub SmartDashboard()

Dim zakres As Range
Dim dane As Variant
Dim ws As Worksheet
Dim pc As PivotCache
Dim pt As PivotTable
Dim ch As ChartObject
Dim scData As SlicerCache
Dim scKlient As SlicerCache

Dim nazwaData As String
Dim nazwaKlient As String
Dim i As Long

Set zakres = Selection
dane = zakres.Value

' nowy arkusz
Set ws = Worksheets.Add

' pivot cache
Set pc = ActiveWorkbook.PivotCaches.Create( _
    SourceType:=xlDatabase, _
    SourceData:=zakres)

' tabela przestawna 
Set pt = pc.CreatePivotTable( _
    TableDestination:=ws.Range("A3"))

' nazwy kolumn
nazwaData = dane(1, 1)
nazwaKlient = dane(1, 2)

With pt
    
    ' Data › wiersze
    .PivotFields(nazwaData).Orientation = xlRowField
    
    ' Klient › filtr
    .PivotFields(nazwaKlient).Orientation = xlPageField
    
    ' wartości 
    For i = 3 To UBound(dane, 2)
        If IsNumeric(dane(2, i)) Then
            .AddDataField .PivotFields(dane(1, i)), "Suma " & dane(1, i), xlSum
        End If
    Next i

End With

'  wykres 
Set ch = ws.ChartObjects.Add(Left:=300, Top:=30, Width:=500, Height:=300)

With ch.Chart
    .SetSourceData pt.TableRange1
    .ChartType = xlColumnClustered
    
    ' tytuł wykresu
    .HasTitle = True
    .ChartTitle.Text = "Analiza danych sprzedażowych poszczególnych klientów (suma wartości)"
    
    ' nazwy osi
    .Axes(xlCategory).HasTitle = True
    .Axes(xlCategory).AxisTitle.Text = nazwaData
    
    .Axes(xlValue).HasTitle = True
    .Axes(xlValue).AxisTitle.Text = "Suma wartości"
    
    ' legenda
    .HasLegend = True

End With

'  fragmentator klienta 
Set scKlient = ActiveWorkbook.SlicerCaches.Add(pt, nazwaKlient)
scKlient.Slicers.Add ws, , "Klient", "Wybierz klienta", _
    ch.Top, ch.Left + ch.Width + 20, 150, 200

'  timeline 
On Error Resume Next
Set scData = ActiveWorkbook.SlicerCaches.Add2(pt, nazwaData, , xlTimeline)

With scData.Slicers(1)
    .Top = ch.Top + ch.Height + 10
    .Left = ch.Left
    .Width = ch.Width
    .Height = 80
End With

On Error GoTo 0

' komunikat dla użytkownika
MsgBox "An automated dashboard has been successfully created." & vbCrLf & vbCrLf & _
       "1. A pivot table has been generated on the left side of the worksheet:" & vbCrLf & _
       "   • The first column (" & nazwaData & ") is used as row labels." & vbCrLf & _
       "   • The second column (" & nazwaKlient & ") is used as a report filter." & vbCrLf & _
       "   • All remaining numeric columns have been aggregated (sum)." & vbCrLf & vbCrLf & _
       "2. A column chart has been created to visualize the aggregated data:" & vbCrLf & _
       "   • X-axis: " & nazwaData & vbCrLf & _
       "   • Y-axis: aggregated values." & vbCrLf & vbCrLf & _
       "3. Interactive filters have been added:" & vbCrLf & _
       "   • Client slicer (located on the right)." & vbCrLf & _
       "   • Timeline filter for date selection (located below the chart)." & vbCrLf & vbCrLf & _
       "Use the slicers to dynamically filter and explore the data.", _
       vbInformation, "Dashboard Created"
End Sub
